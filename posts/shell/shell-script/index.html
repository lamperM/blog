    统计代码量

    使用到的命令包含: find, wc, xargs, sort 等

    列出所有的文件及其代码行数, 只统计.c 和.h, 过滤./scripts目录.

    1find -name '*.[c|h]' ! -path './scripts/*' | xargs  wc -l

    +将内容按照代码行数降序排列

    1find -name '*.[c|h]' ! -path './scripts/*' | xargs  wc -l | sort -rn

    若仅列出总的代码行数, 去除空行

    1(find ./ -name '*.[c|h]' -print0 | xargs -0 cat) | sed '/^\s*$/d' | wc -l

    删除目录下所有的可执行文件

    1find . -maxdepth 1 -executable -type f | xargs rm

    带颜色的输出

    使用ANSI escape code

    1Black        0;30     Dark Gray     1;30
2Red          0;31     Light Red     1;31
3Green        0;32     Light Green   1;32
4Brown/Orange 0;33     Yellow        1;33
5Blue         0;34     Light Blue    1;34
6Purple       0;35     Light Purple  1;35
7Cyan         0;36     Light Cyan    1;36
8Light Gray   0;37     White         1;37

    Code example:

     1#!/bin/bash
 2
 3RED='\033[0;31m'
 4GREEN='\033[0;32m'
 5YELLOW='\033[1;33m'
 6BLUE='\033[0;34m'
 7CYAN='\033[0;36m'
 8NC='\033[0m' # No Color
 9
10echo -e "${YELLOW}HELLO, YELLOW${NC}"
11echo -e "${GREEN}HELLO, GREEN${NC}"
12echo -e "${RED}HELLO, RED${NC}"
13echo -e "${BLUE}HELLO, BLUE${NC}"
14echo -e "${CYAN}HELLO, CYAN${NC}"
15
16#########################################################
17# generic functions #####################################
18
19function ERROR(){
20    echo -e "${RED}[error] $*${NC}";
21    exit 1
22}
23
24function INFO {
25    echo -e "${BLUE}[info] $*${NC}";
26}
27
28function WARN {
29    echo -e "${YELLOW}[warn] $*${NC}";
30}
31
32function LOG {
33    echo -e "${GREEN}[log] $*${NC}" >> $LOG
34}
35
36INFO "This is an infomation"
37WARN  "This is a log"
