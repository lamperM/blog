    常用Shell命令的注意事项

    echo的参数

    1# echo不自动换行
2echo -n "some-string"
3
4# echo转义特殊字符
5echo    "some-string\n" # 不转义\
6echo -e "some-string\n" # 正确转义换行

    Zsh中，echo默认带-e，bash中则不是。 所以说，如果输出的字符串有转义字符，不管要不要转义，都显式指定一下。

    • -e 强制转义
    • -E 强制不转义

    shift 操作参数

    shift命令将参数左移，应用场景:

     1# 场景1：提取第二个参数及后面的所有
 2# ./shell qemu-system-aarch64 -machine raspi3b -smp
 3qemu=$1
 4shift  # 默认左移1
 5qemu_opt=$@
 6
 7
 8## 场景2: 处理多参数且位置不定的情况
 9while [ "$#" -gt 0 ]; do
10  case "$1" in
11    -f|--file)
12      file="$2"
13      shift 2
14      ;;
15    -d|--directory)
16      directory="$2"
17      shift 2
18      ;;
19    -m|--mode)
20      mode="$2"
21      shift 2
22      ;;
23    *)
24      echo "Unknown option: $1"
25      exit 1
26      ;;
27  esac
28done

    Shellscript的编写技巧

    1. Bash脚本正式语句开始前加 set -e, 使得任何命令返回非0时立即退出整个Shell脚本。
    2. 规定分隔符。默认Shell会采用空格、tab、换行都作为分隔符， 有时需要屏蔽某些，仅使用空格
      1export IFS=' ' # 启用空格分隔only
2flag="false"
3for str in $qemu_ver;
4do
5  # do something
6done
7unset IFS  # 恢复默认

    统计代码量

    列出所有的文件及其代码行数, 只统计.c 和.h, 过滤./scripts目录.

    1find -name '*.[c|h]' ! -path './scripts/*' | xargs  wc -l

    +将内容按照代码行数降序排列

    1find -name '*.[c|h]' ! -path './scripts/*' | xargs  wc -l | sort -rn

    若仅列出总的代码行数, 去除空行

    1(find ./ -name '*.[c|h]' -print0 | xargs -0 cat) | sed '/^\s*$/d' | wc -l

    删除目录下所有的可执行文件

    1find . -maxdepth 1 -executable -type f | xargs rm

    带颜色的输出

    使用ANSI escape code

    1Black        0;30     Dark Gray     1;30
2Red          0;31     Light Red     1;31
3Green        0;32     Light Green   1;32
4Brown/Orange 0;33     Yellow        1;33
5Blue         0;34     Light Blue    1;34
6Purple       0;35     Light Purple  1;35
7Cyan         0;36     Light Cyan    1;36
8Light Gray   0;37     White         1;37

    Code example:

     1#!/bin/bash
 2
 3RED='\033[0;31m'
 4GREEN='\033[0;32m'
 5YELLOW='\033[1;33m'
 6BLUE='\033[0;34m'
 7CYAN='\033[0;36m'
 8NC='\033[0m' # No Color
 9
10echo -e "${YELLOW}HELLO, YELLOW${NC}"
11echo -e "${GREEN}HELLO, GREEN${NC}"
12echo -e "${RED}HELLO, RED${NC}"
13echo -e "${BLUE}HELLO, BLUE${NC}"
14echo -e "${CYAN}HELLO, CYAN${NC}"
15
16#########################################################
17# generic functions #####################################
18
19function ERROR(){
20    echo -e "${RED}[error] $*${NC}";
21    exit 1
22}
23
24function INFO {
25    echo -e "${BLUE}[info] $*${NC}";
26}
27
28function WARN {
29    echo -e "${YELLOW}[warn] $*${NC}";
30}
31
32function LOG {
33    echo -e "${GREEN}[log] $*${NC}" >> $LOG
34}
35
36INFO "This is an infomation"
37WARN  "This is a log"
