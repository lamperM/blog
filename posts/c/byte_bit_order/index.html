     1// Bit reverse
 2unsigned char
 3bit_reverse(unsigned char x)
 4{
 5  unsigned char newx = 0;
 6  for (int i = 0; i < 8; i++) {
 7    newx |= (((x >> i) & 1) << (7-i));
 8  }
 9  return newx;
10}

     1// Bit reverse(faster)
 2// 碟式交换法
 3unsigned char
 4bit_reverse_faster(unsigned char x)
 5{
 6  x = (x<<4) | (x>>4); // [ 5678 1234 ]
 7  x = ((x<<2)&0xcc) | ((x>>2)&0x33); // [ 78 56 34 12 ]
 8  x = ((x<<1)&0xaa) | ((x>>1)&0x55); // [ 8 7 6 5 4 3 2 1 ]
 9  return x;
10}

     1int
 2to_binary_str(unsigned long val, char *bin_str, int *len)
 3{
 4  char *p, *q;
 5  int cnt = 0;
 6
 7  while (val) {
 8    char tmp = val & 1;
 9    bin_str[cnt++] = tmp+'0';
10    val = val >> 1;
11  }
12
13  p = bin_str, q = bin_str+cnt-1;
14  while (p < q) {
15    char tmp = *p;
16    *p = *q;
17    *q = tmp;
18    p++, q--;
19  }
20
21  bin_str[cnt] = 0;
22  *len = cnt;
23  return 0;
24}
